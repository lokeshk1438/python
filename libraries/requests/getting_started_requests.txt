requests:
	import request as req
	
	getdata = req.get(r'https://jsonplaceholder.typicode.com/users')
	
	payload={
    'id' : 9,
    'username' : 'Delphine'
	}
	
	getdata = req.get(r'https://jsonplaceholder.typicode.com/users', params=payload)
